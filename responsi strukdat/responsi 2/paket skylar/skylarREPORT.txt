#include <iostream>
using namespace std;

/*
 * TODO 1
 * Buatlah sebuah fungsi rekursif yang menerima sebuah bilangan bulat n
 * dan mengembalikan bilangan fibonacci ke-n (urutan fibonacci dimulai dari 0).
 */
int fibonacci(int n) {
    if (n <= 1)
        return n;
    else
        return fibonacci(n - 1) + fibonacci(n - 2);
}


main() {
    int n;
    cin >> n;
    cout << fibonacci(n);
}
